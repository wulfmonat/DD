  #ui { position:absolute; top:8px; left:8px; background:rgba(0,0,0,0.7); padding:10px; border:1px solid #444; border-radius:6px; font-size:13px; line-height:1.4; }
  #ui b { color:#ffeb3b; }
  canvas { display:block; margin:0 auto; background:#222; image-rendering:pixelated; }
  #help { position:absolute; bottom:8px; left:8px; background:rgba(0,0,0,0.6); padding:6px 10px; border-radius:4px; font-size:12px; }

<body>
<div id="ui"></div>
<canvas id="game" width="960" height="640"></canvas>
<div id="help">WASD/Arrows: move | SPACE: collect | E: sell all | Z: Zone2 | R: respawn | M: debug masks</div>
<script>
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const ui = document.getElementById('ui');

// --- CONFIG ---
const MUSHROOM_TYPES = [
  {id:'yellow', name:'Yellow Morel', value:5, color:'#ffd700', weight:25},
  {id:'black', name:'Black Morel', value:8, color:'#333', weight:20},
  {id:'false', name:'False Morel', value:1, color:'#b94e48', weight:15},
  {id:'chant1', name:'Chanterelle', value:3, color:'#ff9800', weight:3.33},
  {id:'chant2', name:'Chanterelle', value:3, color:'#ff9800', weight:3.33},
  {id:'chant3', name:'Chanterelle', value:3, color:'#ff9800', weight:3.33},
  {id:'bolete1', name:'King Bolete', value:4, color:'#8d6e63', weight:3.33},
  {id:'bolete2', name:'King Bolete', value:4, color:'#8d6e63', weight:3.33},
  {id:'shiitake', name:'Shiitake', value:4, color:'#795548', weight:3.33},
  {id:'rg1', name:'Green Russula', value:2, color:'#7cb342', weight:3.33},
  {id:'rg2', name:'Green Russula', value:2, color:'#7cb342', weight:3.33},
  {id:'rg3', name:'Green Russula', value:2, color:'#7cb342', weight:3.33},
  {id:'rs1', name:'Sickener', value:1, color:'#e53935', weight:3.33},
  {id:'rs2', name:'Sickener', value:1, color:'#e53935', weight:3.33},
  {id:'rs3', name:'Sickener', value:1, color:'#e53935', weight:3.33},
];

const SPAWN_COUNT = 40;
let zone = 1;
let waders = true; // forced ON for testing
let muckBoots = true; // forced ON for testing
let showMasks = true;
let cash = 0;

// Maps
const maps = {
  1: { img:new Image(), water:new Image(), swamp:new Image() },
  2: { img:new Image(), water:new Image(), swamp:new Image() }
};
maps[1].img.src = 'Map_1.png';
maps[1].water.src = 'Water_Puddles_1.png';
maps[1].swamp.src = 'Swamp_Puddles_1.png';
maps[2].img.src = 'Map_2.png';
maps[2].water.src = 'Water_Puddles_1.png'; // reuse if no zone2 specific
maps[2].swamp.src = 'Swamp_Puddles_1.png';

let mushrooms = [];
let inventory = {yellow:0, black:0, false:0, others:0, value:0};
let spawnedCounts = {yellow:0, black:0, false:0, others:0};

const player = { x:480, y:320, w:24, h:24, speed:2.2, dir:'down' };

const keys = {};
window.addEventListener('keydown', e=>{ keys[e.key.toLowerCase()]=true; if([' ','arrowup','arrowdown'].includes(e.key)) e.preventDefault(); });
window.addEventListener('keyup', e=>{ keys[e.key.toLowerCase()]=false; });

function weightedPick() {
  const total = MUSHROOM_TYPES.reduce((s,t)=>s+t.weight,0);
  let r = Math.random()*total;
  for(const t of MUSHROOM_TYPES){ r-=t.weight; if(r<=0) return t; }
  return MUSHROOM_TYPES[0];
}

function spawnMushrooms() {
  mushrooms = [];
  spawnedCounts = {yellow:0, black:0, false:0, others:0};
  inventory = {yellow:0, black:0, false:0, others:0, value:0};
  for(let i=0;i<SPAWN_COUNT;i++){
    const type = weightedPick();
    const m = {
      x: 60 + Math.random()*(canvas.width-120),
      y: 60 + Math.random()*(canvas.height-120),
      type,
      collected:false
    };
    mushrooms.push(m);
    if(type.id==='yellow') spawnedCounts.yellow++;
    else if(type.id==='black') spawnedCounts.black++;
    else if(type.id==='false') spawnedCounts.false++;
    else spawnedCounts.others++;
  }
}

function isBlocked(x,y){
  const map = maps[zone];
  // sample water
  if(map.water.complete && map.water.width){
    const sx = Math.floor(x / canvas.width * map.water.width);
    const sy = Math.floor(y / canvas.height * map.water.height);
    // quick check via offscreen canvas
    // for simplicity, skip pixel perfect if gear forced on
    if(!waders){ /* would block */ }
  }
  if(!muckBoots){ /* would block swamp */ }
  return false;
}

function update() {
  let dx=0, dy=0;
  if(keys['w']||keys['arrowup']){ dy=-1; player.dir='up'; }
  if(keys['s']||keys['arrowdown']){ dy=1; player.dir='down'; }
  if(keys['a']||keys['arrowleft']){ dx=-1; player.dir='left'; }
  if(keys['d']||keys['arrowright']){ dx=1; player.dir='right'; }
  if(dx||dy){
    const nx = player.x + dx*player.speed;
    const ny = player.y + dy*player.speed;
    if(nx>20 && nx<canvas.width-20 && ny>20 && ny<canvas.height-20 && !isBlocked(nx,ny)){
      player.x = nx; player.y = ny;
    }
  }
  if(keys[' ']){
    // collect
    for(const m of mushrooms){
      if(!m.collected){
        const d = Math.hypot(m.x-player.x, m.y-player.y);
        if(d<22){
          m.collected=true;
          inventory.value += m.type.value;
          if(m.type.id==='yellow') inventory.yellow++;
          else if(m.type.id==='black') inventory.black++;
          else if(m.type.id==='false') inventory.false++;
          else inventory.others++;
          break;
        }
      }
    }
    keys[' ']=false;
  }
  if(keys['e']){ cash += inventory.value; inventory = {yellow:0, black:0, false:0, others:0, value:0}; keys['e']=false; }
  if(keys['z']){ zone=2; spawnMushrooms(); keys['z']=false; }
  if(keys['r']){ spawnMushrooms(); keys['r']=false; }
  if(keys['m']){ showMasks=!showMasks; keys['m']=false; }
}

function draw() {
  const map = maps[zone];
  ctx.clearRect(0,0,canvas.width,canvas.height);
  if(map.img.complete) ctx.drawImage(map.img,0,0,canvas.width,canvas.height);
  else { ctx.fillStyle='#2e3d2e'; ctx.fillRect(0,0,canvas.width,canvas.height); }

  if(showMasks){
    if(map.water.complete){ ctx.globalAlpha=0.35; ctx.drawImage(map.water,0,0,canvas.width,canvas.height); }
    if(map.swamp.complete){ ctx.globalAlpha=0.25; ctx.fillStyle='#9c27b0'; ctx.globalAlpha=0.25; ctx.drawImage(map.swamp,0,0,canvas.width,canvas.height); }
    ctx.globalAlpha=1;
  }

  // mushrooms
  for(const m of mushrooms){
    if(m.collected) continue;
    ctx.beginPath();
    ctx.arc(m.x, m.y, 10, 0, Math.PI*2);
    ctx.fillStyle = m.type.color;
    ctx.fill();
    ctx.strokeStyle = '#000'; ctx.lineWidth=2; ctx.stroke();
    // small stem
    ctx.fillStyle='#d7ccc8'; ctx.fillRect(m.x-2, m.y+6,4,6);
  }

  // player - simple 48x48 placeholder
  ctx.fillStyle = '#2196f3';
  ctx.fillRect(player.x-12, player.y-16, 24, 32);
  ctx.fillStyle = '#ffcc80';
  ctx.fillRect(player.x-8, player.y-24, 16, 10); // head
  // direction poke
  ctx.fillStyle='#fff';
  if(player.dir==='up') ctx.fillRect(player.x-2, player.y-28,4,6);
  if(player.dir==='down') ctx.fillRect(player.x-2, player.y+16,4,6);
  if(player.dir==='left') ctx.fillRect(player.x-18, player.y-4,6,4);
  if(player.dir==='right') ctx.fillRect(player.x+12, player.y-4,6,4);

  // UI
  const totalSpawned = spawnedCounts.yellow+spawnedCounts.black+spawnedCounts.false+spawnedCounts.others;
  const morelPct = Math.round((spawnedCounts.yellow+spawnedCounts.black+spawnedCounts.false)/SPAWN_COUNT*100);
  ui.innerHTML = `
    <b>Morel Forager v5.9v</b> | Zone ${zone} | Cash $${cash}<br>
    Waders: ${waders?'ON':'OFF'} | Muck Boots: ${muckBoots?'ON':'OFF'}<br><br>
    <b>SPAWNED (${totalSpawned}/${SPAWN_COUNT}) - ${morelPct}% morels</b><br>
    Yellow: ${spawnedCounts.yellow} | Black: ${spawnedCounts.black} | False: ${spawnedCounts.false} | Others: ${spawnedCounts.others}<br><br>
    <b>COLLECTED</b><br>
    Yellow: ${inventory.yellow} | Black: ${inventory.black} | False: ${inventory.false} | Others: ${inventory.others}<br>
    Bag Value: $${inventory.value}
  `;
}

function loop(){ update(); draw(); requestAnimationFrame(loop); }

maps[1].img.onload = ()=>{ spawnMushrooms(); loop(); };
// fallback if image fails
setTimeout(()=>{ if(mushrooms.length===0){ spawnMushrooms(); loop(); } },1000);
</script>
</body>
</html>
